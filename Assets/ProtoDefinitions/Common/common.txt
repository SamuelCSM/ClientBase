// 通用类定义文件

// 玩家信息通用类
message PlayerInfo {
    int64 PlayerId = 1;  // 玩家ID
    string PlayerName = 2;  // 玩家名称
    int32 Level = 3;  // 等级
    int64 Exp = 4;  // 经验值
    int32 VipLevel = 5;  // VIP等级
}

// 物品信息通用类
message ItemInfo {
    int32 ItemId = 1;  // 物品ID
    int32 Count = 2;  // 数量
    int32 Quality = 3;  // 品质
}

// 位置信息通用类
message Position {
    float X = 1;  // X坐标
    float Y = 2;  // Y坐标
    float Z = 3;  // Z坐标
}
